The following definition describes the ILoggable interface that should be implemented by loggable data structures. Timestamp holds the exact time point of the event to be logged. 

using System;
        namespace Logging
        {
            // This interface represent log data.
            public interface ILoggable
            {
                // Gets or sets the time of the event
                DateTime Timestamp { get; set; }
            }
        }
